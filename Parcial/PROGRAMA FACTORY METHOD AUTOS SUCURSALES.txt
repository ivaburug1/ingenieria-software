Crear programa en Console App (.NET Framework)


Programa:
Auto.cs:
namespace FactoryMethod
{
    public abstract class Auto
    {
        protected string _descripcion;
        protected string _modalidad;

        public override string ToString()
        {
            return string.Format("- Venta '{0}' bajo modalidad: {1}", _descripcion, _modalidad);
        }
    }
}

AutoContado.cs:
namespace FactoryMethod
{
    public class AutoContado : Auto
    {
        public AutoContado(string modalidad)
        {
            _descripcion = "Venta de auto al contado";
            _modalidad = modalidad;
        }
    }
}

Vendedor.cs:
namespace FactoryMethod
{
    public abstract class Vendedor
    {
        public abstract Auto CrearOferta(string tipo);
    }
}

VendedorSucursalA.cs:
namespace FactoryMethod
{
    public class VendedorSucursalA : Vendedor
    {
        public override Auto CrearOferta(string tipo)
        {
            if (tipo == "contado")
            {
                return new AutoContado("Sucursal A");
            }
            else if (tipo == "financiado")
            {
                return new AutoFinanciado("Sucursal A");
            }
            else
            {
                return null;
            }
        }
    }
}
